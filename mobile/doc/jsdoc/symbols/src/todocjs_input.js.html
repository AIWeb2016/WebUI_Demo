1 /**
  2  * 输入框组件绑定事件
  3  * @fileOverview
  4  * @author kangbk
  5  * @time 2015-8-11
  6  * @version 1.0
  7  */
  8 $(function() {
  9 	/**
 10 	 * @constructor input
 11 	 * @desc 输入框组件绑定js事件
 12 	 * @name input
 13 	 *
 14 	 */
 15 	//清除按钮绑定清除事件
 16 	/**
 17 	 * 清除按钮绑定清除事件
 18 	 */
 19 	$(".js-ui-clear").each(function() {
 20 		$(this).bind("click", function() {
 21 			$(this).prev().val("");
 22 			$(this).prev().focus();
 23 		});
 24 	});
 25 	//查看密码，单击查看文本，再次单击显示密文
 26 	/**
 27 	 * 密码输入框绑定显示，隐藏按钮
 28 	 */
 29 	$(".js-ui-look").bind("click", function() {
 30 		var _ = $(this).parent().find("input");
 31 		if (_.attr("type") == "text") {
 32 			_.attr("type", "password");
 33 			$(this).find("span").css("background", "url(img/look.png)");
 34 			$(this).prev().focus();
 35 		} else {
 36 			_.attr("type", "text");
 37 			$(this).find("span").css("background", "url(img/look-on.png)");
 38 			$(this).prev().focus();
 39 		}
 40 	});
 41 	//给文本域绑定输入变化事件，并实时更新还可输入字符数量
 42 	/**
 43 	 * 给文本域绑定输入变化事件，并实时更新还可输入字符数量
 44 	 */
 45 	$(".ui-textarea").bind('input propertychange', function() {
 46 		var len = parseInt($(this).val().length);
 47 		var maxlen = parseInt($(this).attr("maxlength"));
 48 		var mulit = maxlen - len;
 49 		if (len < maxlen || len == maxlen) {
 50 			$('#JS_lastNum').html(maxlen - len);
 51 		} else {
 52 			$('#JS_lastNum').html(0);
 53 		}
 54 		if (mulit > 10) {
 55 			$(".ui-textarea-tip").hide();
 56 		} else {
 57 			$(".ui-textarea-tip").show();
 58 		}
 59 	});
 60 	/**
 61 	 * 自适应高度增加行数
 62 	 * @param {Object} element 当前元素
 63 	 * @param {Object} minHeight 最低高度
 64 	 */
 65 	function autoHeight(element, minHeight) {
 66 		// 计算因边框和轮廓产生的高度差异
 67 		var outerHeight = parseInt(window.getComputedStyle(el).height, 10);
 68 		var diff = outerHeight - el.clientHeight;
 69 		// 设置高度为0以防需要收缩（高度）
 70 		el.style.height = 0;
 71 		// 设置正确的高度
 72 		el.style.height = Math.max(minHeight, el.scrollHeight + diff) + 'px';
 73 	}
 74 	var _textarea = $(".js-textarea");
 75 	// 迭代本页所有的文本域
 76 	for (var i = 0, l = _textarea.length; i < l; i++) {
 77 		var el = _textarea[i];
 78 		// 如果文本域有边距，我们需要设置box-sizing: border-box
 79 		el.style.boxSizing = el.style.mozBoxSizing = 'border-box';
 80 		// 我们不需要滚动条，不是么？ :)
 81 		el.style.overflowY = 'hidden';
 82 		// 通过"rows"属性初始化的最小高度
 83 		var minHeight = el.scrollHeight;
 84 		el.addEventListener('input', function() {
 85 			autoHeight(el, minHeight);
 86 		});
 87 		// 当窗口大小改变时，我们需要重新调整高度（例如方向变化）
 88 		window.addEventListener('resize', function() {
 89 			autoHeight(el, minHeight);
 90 		});
 91 		// 我们调整初始内容的高度
 92 		autoHeight(el, minHeight);
 93 	}
 94 });
