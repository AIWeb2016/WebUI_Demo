1 /**
  2  * 顶部加载组件--实现顶部加载的效果，可手动控制加载的开始和结束
  3  *
  4  * @fileOverview

  8 ;(function($) {
  9 	/**
 10 	 * 顶部加载
 11 	 *
 12 	 * @constructor Progress
 13 	 * @desc 顶部加载
 14 	 * @memberOf jQuery
 15 	 * @name progress
 16 	 * 
 17 	 * @return {Object} 返回 progress 对象的一个实例
 18 	 * @example
 19 	 * var progress = $('#btn').progress();
 20 	 *
 21 	 */
 22 	$.fn.progress = function() {
 23 		return new Progress(this);
 24 	};
 25 	var Progress = function(element) {
 26 		this.el = $(element);
 27 		this.init();
 28 	}
 29 	/**
 30 	 * progress 名字空间
 31 	 * @namespace
 32 	 * @memberOf jQuery
 33 	 * @name progress
 34 	 */
 35 	Progress.prototype = 
 36 	/**
 37 	 * @lends jQuery.progress
 38 	 */
 39 	{
 40 		/**
 41 		 * 初始化
 42 		 */
 43 		init: function() {
 44 			this.creatDiv();
 45 		},
 46 		/** 
 47 		 * 构造页面显示div
 48 		 */
 49 		creatDiv: function() {
 50 			var _load = $("<div>").addClass("ui-load-top").appendTo(this.el);
 51 			var _line = $("<div>").addClass("ui-load-line").appendTo(_load);
 52 			_load.hide();
 53 		},
 54 		/** 
 55 		 * 实现开始的效果
 56 		 */
 57 		start: function() { //打开进度条
 58 			var load = $(".ui-load-top");
 59 			var _lineWidth = 0;
 60 			var _loadWidth = parseInt($(".ui-load-top").width());
 61 			var line = $(".ui-load-line");
 62 			var item = (_lineWidth / _loadWidth) * 100;
 63 			load.show();
 64 			/*
 65 			 * 进度条根据时间变化
 66 			 */
 67 			setInterval(function() {
 68 				if (item < 30) {
 69 					item += 1.3;
 70 					line.css("width", item + "%");
 71 				} else {
 72 					return;
 73 				}
 74 			}, 20);
 75 			setInterval(function() {
 76 				var load = $(".ui-load-top");
 77 				var _lineWidth = parseInt($(".ui-load-line").width());
 78 				var _loadWidth = parseInt($(".ui-load-top").width());
 79 				var line = $(".ui-load-line");
 80 				var item = (_lineWidth / _loadWidth) * 100;
 81 				if (item < 85) {
 82 					item += 0.3;
 83 					line.css("width", item + "%");
 84 				} else {
 85 					return;
 86 				}
 87 			}, 50);
 88 		},
 89 		/** 
 90 		 * 实现结束的效果
 91 		 */
 92 		done: function() {
 93 			var load = $(".ui-load-top");
 94 			var _lineWidth = parseInt($(".ui-load-line").width());
 95 			var _loadWidth = parseInt($(".ui-load-top").width());
 96 			var line = $(".ui-load-line");
 97 			var item = (_lineWidth / _loadWidth) * 100;
 98 			item = Math.floor(item);
 99 			for (; item < 101; item++) {
100 				line.css("width", item + "%");
101 				if (item >= 100) {
102 					setTimeout(function() {
103 						load.hide();
104 					}, 700);
105 					return;
106 				}
107 			}
108 		}
109 	};
110 })(jQuery)
