 23 	 * @return {Object} 返回drag对象的一个实例

 34 	//绑定到jquery上]
 35 	$.fn.drag = function(options) {
 36 		return new Drag(this, options);
 37 	}

 39 	var Drag = function(element, options) {
 40 		var _ = this;
 41 		_.el = $(element);
 42 		_.insertDOM = false;
 43 		_.loading = false;
 44 		_.init(options);
 45 	}

 47 	/**
 48 	 * drag 名字空间
 49 	 * @namespace
 50 	 * @memberOf jQuery
 51 	 * @name drag
 52 	 */
 53 	Drag.prototype =
 54 		/**
 55 		 * @lends jQuery.drag
 56 		 */
 57 		{
 58 			/**
 59 			 * 初始化，拼装数据和模版加载到容器
 60 			 * 
 61 			 * @param {Object} option 参数对象
 62 			 */
 63 			init: function(options) {
 64 				var _ = this;
 65 				_.opts = $.extend({}, {
 66 					domUp: {
 67 						domClass: 'drag-up',
 68 						domRefresh: '<section class="fn-layout-vcenter"><div class="drag-refresh"><img class="drag-img" width="50" height="50" src="img/down.png"/>下拉刷新</div></section>',
 69 						domUpdate: '<section class="fn-layout-vcenter"><div class="drag-update"><img class="drag-img" width="50" height="50" src="img/top.png"/>释放更新</div></section>',
 70 						domLoad: '<section class="fn-layout-vcenter"><div class="drag-load"><span class="loading"></span>加载中...</div></section>',
 71 						domSuccess: '<section class="fn-layout-vcenter"><div class="drag-result"><img class="drag-img" width="32" height="32" src="img/load-success.png" />刷新成功</div></section>'
 72 					},
 73 					domDown: {
 74 						domClass: 'drag-down',
 75 						domRefresh: '<div class="drag-refresh"></div>',
 76 						domUpdate: '<div class="drag-update"></div>',
 77 						domLoad: '<section class="fn-layout-vcenter"><div class="drag-load"><img class="drag-img" width="50" height="50" src="img/ui-loading.gif"/></span>加载中...</div></section>'
 78 					},
 79 					distance: 92,
 80 					refreshURL: '', //刷新的url
 81 					loadMoreURL: '', //加载的url
 82 					loadMoreText: '', //底部提示文字,如果没有则没有底部按钮
 83 					flag: true,
 84 					loadUpFn: function(_) {
 85 						$.ajax({
 86 							type: 'GET',
 87 							url: _.opts.refreshURL,
 88 							dataType: 'json',
 89 							success: function(data) {
 90 								var result = _.opts.model(data.lists);
 91 								$('.lists').html(result);
 92 								$(".ui-btn-more").show();
 93 								_.opts.flag = true;
 94 								_.loadsuccess();
 95 								setTimeout(function() {
 96 									_.resetload();
 97 								}, 1200);
 98 
 99 							},
100 							error: function(xhr, type) {
101 								alert('Ajax error!');
102 								_.resetload();
103 							}
104 						});
105 					},
106 					loadDownFn: function(_) {
107 						$.ajax({
108 							type: 'GET',
109 							url: _.opts.loadMoreURL,
110 							dataType: 'json',
111 							success: function(data) {
112 								var result = '';
113 								if (data.lists.length < 1) {
114 									$(".ui-btn-more").hide();
115 									result = '<span class="ui-tip-down">没有更多数据了</span>';
116 									if (_.opts.flag == true) {
117 										$('.lists').append(result);
118 										_.opts.flag = false;
119 									}
120 								} else {
121 									result = _.opts.model(data.lists);
122 									$('.lists').append(result);
123 								}
124 								_.resetload();
125 							},
126 							error: function(xhr, type) {
127 								alert('Ajax error!');
128 								_.resetload();
129 							}
130 						});
131 					},
132 					model: function(data) {
133 						var result = '';
134 						for (var i = 0; i < data.length; i++) {
135 							result += '<a class="item opacity">' + '<img src="' + data[i].pic + '" alt="">' + '<h3 href="' + data[i].link + '" >' + data[i].title + '</h3>' + '<span class="date">' + data[i].date + '</span>' + '</a>';
136 						}
137 						return result;
138 					}
139 				}, options);
140 				if (_.opts.loadMoreText != '') {
141 					//添加加载剩下几条按钮

158 						dragTouchmove(e, _);
159 					}
160 				});
161 				_.el.on('touchend', function(e) {
162 					if (!_.loading) {
163 						dragTouchend(_);
164 					}
165 				});
166 			},

206 		if (!e.touches) {
207 			e.touches = e.originalEvent.touches;
208 		}

210 	/** 
211 	 *  触摸开始
212 	 * @param {Event} e - 事件
213 	 * @param {Object} obj - 事件对象
214 	 */
215 	function dragTouchstart(e, obj) {
216 		obj._startY = e.touches[0].pageY; //开始位置
217 		obj._loadHeight = obj.el.height(); //加载提示高度
218 		obj._childrenHeight = obj.el.children().height(); //子类元素高度
219 		obj._scrollTop = obj.el.scrollTop(); //滚动条
220 	};
221 	/** 
222 	 *  移动事件
223 	 * @param {Event} e - 事件
224 	 * @param {Object} obj - 事件对象
225 	 */
226 	function dragTouchmove(e, obj) {
227 		obj._curY = e.touches[0].pageY; //获取Y轴位置
228 		obj._moveY = obj._curY - obj._startY; //获取移动值
229 		//判断是向下还是向上拖动 大于0   向下拖动  小于0  向上推动
230 		if (obj._moveY > 0) {
231 			obj.direction = 'down';
232 		} else if (obj._moveY < 0) {
233 			obj.direction = 'up';
234 		}
235 
236 		var _absMoveY = Math.abs(obj._moveY); //取移动距离的绝对值，获得移动的实际大小
237 		// 上方刷新
238 		if (obj.opts.loadUpFn != '' && obj._scrollTop <= 0 && obj.direction == 'down') {
239 			e.preventDefault(); //禁止拖动时的默认事件
240 			if (!obj.insertDOM) {
241 				obj.el.prepend('<div class="' + obj.opts.domUp.domClass + '"></div>');
242 				obj.insertDOM = true; //修改状态为可以添加元素
243 			}
244 			obj.$domUp = $('.' + obj.opts.domUp.domClass);
245 			dragTransition(obj.$domUp, 0); //css过渡
246 			// 下拉刷新
247 			if (_absMoveY <= obj.opts.distance) {
248 				obj._offsetY = _absMoveY;
249 				obj.$domUp.html('').append(obj.opts.domUp.domRefresh);
250 				// 指定距离 < 下拉距离 < 指定距离*2
251 			} else if (_absMoveY > obj.opts.distance && _absMoveY <= obj.opts.distance * 2) {
252 				obj._offsetY = obj.opts.distance + (_absMoveY - obj.opts.distance) * 0.5;
253 				obj.$domUp.html('').append(obj.opts.domUp.domUpdate);
254 				// 下拉距离 > 指定距离*2
255 			} else {
256 				obj._offsetY = obj.opts.distance + obj.opts.distance * 0.5 + (_absMoveY - obj.opts.distance * 2) * 0.2;
257 			}
258 
259 			obj.$domUp.css({
260 				'height': obj._offsetY
261 			});
262 		}
263 		// 加载更多
264 		if (obj.opts.loadDownFn != '' && obj._childrenHeight <= (obj._loadHeight + obj._scrollTop) && obj.direction == 'up') {
265 			e.preventDefault(); //禁止拖动时的默认事件
266 			if (!obj.insertDOM) {
267 				obj.el.append('<div class="' + obj.opts.domDown.domClass + '"></div>');
268 				obj.insertDOM = true;
269 			}
270 			obj.$domDown = $('.' + obj.opts.domDown.domClass);
271 			dragTransition(obj.$domDown, 0);
272 			// 向上拖动加载更多
273 			if (_absMoveY <= obj.opts.distance) {
274 				obj._offsetY = _absMoveY;
275 				obj.$domDown.html('').append(obj.opts.domDown.domRefresh);
276 				// 指定距离 < 上拉距离 < 指定距离*2
277 			} else if (_absMoveY > obj.opts.distance && _absMoveY <= obj.opts.distance * 2) {
278 				obj._offsetY = obj.opts.distance + (_absMoveY - obj.opts.distance) * 0.5;
279 				obj.$domDown.html('').append(obj.opts.domDown.domUpdate);
280 				// 上拉距离 > 指定距离*2
281 			} else {
282 				obj._offsetY = obj.opts.distance + obj.opts.distance * 0.5 + (_absMoveY - obj.opts.distance * 2) * 0.2;
283 			}
284 
285 			obj.$domDown.css({
286 				'height': obj._offsetY
287 			});
288 			obj.el.scrollTop(obj._offsetY + obj._scrollTop);
289 		}
290 	};
291 	/** 
292 	 *  结束
293 	 * @param {Object} obj - 事件对象
294 	 */
295 	function dragTouchend(obj) {
296 		var _absMoveY = Math.abs(obj._moveY);
297 		if (obj.insertDOM) {
298 			if (obj.direction == 'down') { //向下拖动刷新
299 				obj.$domResult = obj.$domUp;
300 				obj.domLoad = obj.opts.domUp.domLoad;
301 			} else if (obj.direction == 'up') { //向上拖动加载更多
302 				obj.$domResult = obj.$domDown;
303 				obj.domLoad = obj.opts.domDown.domLoad;
304 			}
305 
306 			dragTransition(obj.$domResult, 300); //css过渡
307 			//判读是否拖动距离达到刷新条件或者加载条件
308 			if (_absMoveY > obj.opts.distance) {
309 				//设置刷新的高度和子类高度一样
310 				obj.$domResult.css({
311 					'height': obj.$domResult.children().height()
312 				});
313 				obj.$domResult.html('').append(obj.domLoad); //添加元素到界面
314 				dragCallback(obj); //调用回掉函数，回到原始状态
315 			} else {
316 				//设置高度为0，即不显示效果，不做操作
317 				obj.$domResult.css({
318 					'height': '0'
319 				}).on('webkitTransitionEnd', function() {
320 					obj.insertDOM = false;
321 					$(this).remove();
322 				});
323 			}
324 			obj._moveY = 0;
325 		}
326 	};
327 	/** 
328 	 *  回调函数
329 	 * @param {Object} obj - 事件对象
330 	 */
331 	function dragCallback(obj) {
332 		obj.loading = true;
333 		if (obj.opts.loadUpFn != '' && obj.direction == 'down') {
334 			obj.opts.loadUpFn(obj);
335 		} else if (obj.opts.loadDownFn != '' && obj.direction == 'up') {
336 			obj.opts.loadDownFn(obj);
337 		}
338 	};
