1 /**
  2  * 环形加载
  3  * @author kangbk
  4  * @param {Object} $ 匿名对象
  5  */
  6 (function($) {
  7 	/**
  8 	 * @description 绑定到dom上
  9 	 * @param {Object} options 参数对象
 10 	 * @param {String} fgcolor 填充颜色
 11 	 * @param {String} bgcolor 背景颜色
 12 	 * @param {Number} bgcolor 大小
 13 	 * @param {Number} percent 百分比
 14 	 */
 15 	$.fn.progressCircle = function(options) {
 16 		var settings = $.extend({
 17 			startdegree: 0,//开始地方
 18 			fgcolor: "#556b2f",
 19 			bgcolor: "#eee",
 20 			fill: false,
 21 			dimension: 100,//大小
 22 			percent: 100,//默认百分比为100结束
 23 			animationstep: 1.0, //动画时间
 24 			complete: null,
 25 			bordersize: 2//圆环厚度，默认为2px
 26 		}, options);
 27 		//返回对象
 28 		return this.each(function() {
 29 			var customSettings = ["fgcolor", "bgcolor", "fill", "width", "dimension", "fontsize", "animationstep", "endPercent", "icon", "iconcolor", "iconsize", "border", "startdegree", "bordersize"];
 30 			var customSettingsObj = {};
 31 			var icon = '';
 32 			var percent;
 33 			var endPercent = 0;
 34 			var obj = $(this);
 35 			var fill = false;
 36 			checkDataAttributes(obj);
 37 
 38 			if ($(this).data("total") != undefined && $(this).data("part") != undefined) {
 39 				var total = $(this).data("total") / 100;
 40 
 41 				percent = (($(this).data("part") / total) / 100).toFixed(3);
 42 				endPercent = ($(this).data("part") / total).toFixed(3);
 43 			} else {
 44 				if ($(this).data("percent") != undefined) {
 45 					percent = $(this).data("percent") / 100;
 46 					endPercent = $(this).data("percent");
 47 				} else {
 48 					percent = settings.percent / 100;
 49 				}
 50 			}
 51 			$(this).width(customSettingsObj.dimension + 'px');
 52 
 53 			var size = customSettingsObj.dimension,
 54 				canvas = $('<canvas></canvas>').attr({
 55 					width: size,
 56 					height: size
 57 				}).appendTo($(this)).get(0);
 58 
 59 			var context = canvas.getContext('2d');
 60 
 61 			var dpr = window.devicePixelRatio;
 62 			if (dpr) {
 63 				var $canvas = $(canvas);
 64 				$canvas.css('width', size);
 65 				$canvas.css('height', size);
 66 				$canvas.attr('width', size * dpr);
 67 				$canvas.attr('height', size * dpr);
 68 
 69 				context.scale(dpr, dpr);
 70 			}
 71 			var container = $(canvas).parent();
 72 			var x = size / 2;
 73 			var y = size / 2;
 74 			var degrees = customSettingsObj.percent * 360.0;
 75 			var radians = degrees * (Math.PI / 180);
 76 			var radius = size / 2;
 77 			var startAngle = 2.3 * Math.PI;
 78 			var endAngle = 0;
 79 			var curPerc = customSettingsObj.animationstep === 0.0 ? endPercent : 0.0;
 80 			var curStep = Math.max(customSettingsObj.animationstep, 0.0);
 81 			var circ = Math.PI * 2;
 82 			var quart = Math.PI / 2;
 83 			var fireCallback = true;
 84 			var additionalAngelPI = (customSettingsObj.startdegree / 180) * Math.PI;
 85 
 86 			/**
 87 			 * 检查页面中的参数设置
 88 			 * @param obj
 89 			 */
 90 			function checkDataAttributes(obj) {
 91 				$.each(customSettings, function(index, attribute) {
 92 					if (obj.data(attribute) != undefined) {
 93 						customSettingsObj[attribute] = obj.data(attribute);
 94 					} else {
 95 						customSettingsObj[attribute] = $(settings).attr(attribute);
 96 					}
 97 
 98 					if (attribute == 'fill' && obj.data('fill') != undefined) {
 99 						fill = true;
100 					}
101 				});
102 			}
103 			/**
104 			 * 加载动画
105 			 * @param current 加载百分比
106 			 */
107 			function animate(current) {
108 				context.clearRect(0, 0, canvas.width, canvas.height);
109 				context.beginPath();
110 				context.arc(x, y, radius, endAngle, startAngle, false);
111 				context.lineWidth = customSettingsObj.width + 1;
112 				context.strokeStyle = customSettingsObj.bgcolor;
113 				context.stroke();
114 				if (fill) {
115 					context.fillStyle = customSettingsObj.fill;
116 					context.fill();
117 				}
118 				context.beginPath();
119 				context.arc(x, y, radius, -(quart) + additionalAngelPI, ((circ) * current) - quart + additionalAngelPI, false);
120 
121 				if (customSettingsObj.border == 'outline') {
122 					context.lineWidth = customSettingsObj.bordersize;
123 				} else if (customSettingsObj.border == 'inline') {
124 					context.lineWidth = customSettingsObj.bordersize;
125 				}
126 				context.strokeStyle = customSettingsObj.fgcolor;
127 				context.stroke();
128 				//判读是否加载完成
129 				if (curPerc < endPercent) {
130 					curPerc += curStep;
131 					requestAnimationFrame(function() {
132 						animate(Math.min(curPerc, endPercent) / 100);
133 					}, obj);
134 				}
135 				if (curPerc == endPercent && fireCallback && typeof(options) != "undefined") {
136 					if ($.isFunction(options.complete)) {
137 						options.complete();
138 						fireCallback = false;
139 					}
140 				}
141 			}
142 			animate(curPerc / 100);
143 		});
144 	};
145 }(jQuery));
