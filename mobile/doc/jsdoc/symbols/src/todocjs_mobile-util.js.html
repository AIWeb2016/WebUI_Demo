1 /**
  2  * MobileWeb 通用功能助手，包含常用的 UA 判断、页面适配、search 参数转 键值对。
  3  * 该 JS 应在 head 中尽可能早的引入，减少重绘。
  4  *
  5  * fixScreen 方法根据两种情况适配，该方法自动执行。
  6  *      1. 定宽： 对应 meta 标签写法 -- <meta name="viewport" content="width=750">
  7  *          该方法会提取 width 值，主动添加 scale 相关属性值。
  8  *          注意： 如果 meta 标签中指定了 initial-scale， 该方法将不做处理（即不执行）。
  9  *      2. REM: 不用写 meta 标签，该方法根据 dpr 自动生成，并在 html 标签中加上 data-dpr 和 font-size 两个属性值。
 10  *          该方法约束：IOS 系统最大 dpr = 3，其它系统 dpr = 1，页面每 dpr 最大宽度（即页面宽度/dpr） = 750，REM 换算比值为 16。
 11  *          对应 css 开发，任何弹性尺寸均使用 rem 单位，rem 默认宽度为 视觉稿宽度 / 16;
 12  *              scss 中 $ppr(pixel per rem) 变量写法 -- $ppr: 750px/16/1rem;
 13  *                      元素尺寸写法 -- html { font-size: $ppr*1rem; } body { width: 750px/$ppr; }。
 14 
 15  */
 16 window.mobileUtil = (function(win, doc) {
 17 	var UA = navigator.userAgent,
 18 		isAndroid = /android|adr/gi.test(UA),
 19 		isIos = /iphone|ipod|ipad/gi.test(UA) && !isAndroid, // 据说某些国产机的UA会同时包含 android iphone 字符
 20 		isMobile = isAndroid || isIos;  // 粗略的判断
 21 
 22 	return {
 23 		isAndroid: isAndroid,
 24 		isIos: isIos,
 25 		isMobile: isMobile,
 26 
 27         isNewsApp: /NewsApp\/[\d\.]+/gi.test(UA),
 28 		isWeixin: /MicroMessenger/gi.test(UA),
 29 		isQQ: /QQ\/\d/gi.test(UA),
 30 		isYixin: /YiXin/gi.test(UA),
 31 		isWeibo: /Weibo/gi.test(UA),
 32 		isTXWeibo: /T(?:X|encent)MicroBlog/gi.test(UA),
 33 
 34 		tapEvent: isMobile ? 'tap' : 'click',
 35 
 36 		/**
 37 		 * 缩放页面
 38 		 */
 39 		fixScreen: function() {
 40             var metaEl = doc.querySelector('meta[name="viewport"]'),
 41                 metaCtt = metaEl ? metaEl.content : '',
 42                 matchScale = metaCtt.match(/initial\-scale=([\d\.]+)/),
 43 			    matchWidth = metaCtt.match(/width=([^,\s]+)/);
 44 
 45             if ( !metaEl ) { // REM
 46                 var docEl = doc.documentElement,
 47                     maxwidth = docEl.dataset.mw || 750, // 每 dpr 最大页面宽度
 48                     dpr = isIos ? Math.min(win.devicePixelRatio, 3) : 1,
 49                     scale = 1 / dpr,
 50                     tid;
 51 
 52                 docEl.removeAttribute('data-mw');
 53                 docEl.dataset.dpr = dpr;
 54                 metaEl = doc.createElement('meta');
 55                 metaEl.name = 'viewport';
 56                 metaEl.content = fillScale(scale);
 57                 docEl.firstElementChild.appendChild(metaEl);
 58 
 59                 var refreshRem = function() {
 60                     var width = docEl.getBoundingClientRect().width;
 61                     if (width / dpr > maxwidth) {
 62                         width = maxwidth * dpr;
 63                     }
 64                     var rem = width / 16;
 65                     docEl.style.fontSize = rem + 'px';
 66                 };
 67 
 68                 win.addEventListener('resize', function() {
 69                     clearTimeout(tid);
 70                     tid = setTimeout(refreshRem, 300);
 71                 }, false);
 72                 win.addEventListener('pageshow', function(e) {
 73                     if (e.persisted) {
 74                         clearTimeout(tid);
 75                         tid = setTimeout(refreshRem, 300);
 76                     }
 77                 }, false);
 78 
 79                 refreshRem();
 80             } else if ( isMobile && !matchScale && ( matchWidth && matchWidth[1] != 'device-width' ) ) { // 定宽
 81                 var	width = parseInt(matchWidth[1]),
 82                     iw = win.innerWidth || width,
 83                     ow = win.outerWidth || iw,
 84                     sw = win.screen.width || iw,
 85                     saw = win.screen.availWidth || iw,
 86                     ih = win.innerHeight || width,
 87                     oh = win.outerHeight || ih,
 88                     ish = win.screen.height || ih,
 89                     sah = win.screen.availHeight || ih,
 90                     w = Math.min(iw,ow,sw,saw,ih,oh,ish,sah),
 91                     scale = w / width;
 92 
 93 
 94                 if ( scale < 1 ) {
 95                     metaEl.content = metaCtt + ',' + fillScale(scale);
 96                 }
 97             }
 98 
 99             function fillScale(scale) {
100                 alert('initial-scale=' + scale + ',maximum-scale=' + scale + ',minimum-scale=' + scale + ',user-scalable=no');
101                 return 'initial-scale=' + scale + ',maximum-scale=' + scale + ',minimum-scale=' + scale + ',user-scalable=no';
102             }
103 		},
104 
105 		/**
106 		 * 转href参数成键值对
107 		 * @param href {string} 指定的href，默认为当前页href
108 		 * @returns {object} 键值对
109 		 */
110 		getSearch: function(href) {
111 			href = href || win.location.search;
112 			var data = {},reg = new RegExp( "([^?=&]+)(=([^&]*))?", "g" );
113 			href && href.replace(reg,function( $0, $1, $2, $3 ){
114 				data[ $1 ] = $3;
115 			});
116 			return data;
117 		}
118 	};
119 })(window, document);
