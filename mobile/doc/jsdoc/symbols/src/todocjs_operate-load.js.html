1 /**
  2  * 操作加载组件
  3  *
  4  * @fileOverview

  6  * @version 1.0
  7  */
  8 ;(function($) {
  9 	/*
 10 	 * 绑定为jQuery插件
 11 	 */
 12 	/**
 13 	 * 操作加载
 14 	 *
 15 	 * @constructor Operatingload
 16 	 * @desc 操作加载
 17 	 * @memberOf jQuery
 18 	 * @name operatingload
 19 	 * @param {Object} option 参数对象
 20 	 *
 21 	 * @property {String} tipText - 提示的文字
 22 	 * @property {String} URL - 操作访问的地址
 23 	 * @property {String} param - 前台需要传到后台的参数
 24 	 * @property {String} time - 需要多长时间显示关闭按钮
 25 	 * @property {Function} callback - 提供的回调函数，一共两个参数state(状态)、json(返回的json数据)
 26 	 * @return {Object} 返回 operatingload 对象的一个实例
 27 	 * @example
 28 	 * var operatingload = $('#btn').operatingload({
 29 	 * 	URL:'json/update.json',//访问的地址
 30 		tipText:'这个是测试的问文本！',//提示文本
 31 		callback:function(state,json){
 32 			console.log(JSON.stringify(json));
 33 		}
 34 	 * });
 35 	 *
 36 	 */
 37 	$.fn.operatingload = function(options) {
 38 		return new Operatingload(this, options);
 39 	};
 40 	
 41 	var Operatingload = function(element, options) {
 42 		var _ = this;
 43 		_.el = $(element);
 44 		_.init(options);
 45 	};
 46 	/**
 47 	 * operatingload 名字空间
 48 	 * @namespace
 49 	 * @memberOf jQuery
 50 	 * @name operatingload
 51 	 */
 52 	Operatingload.prototype = 
 53 	/**
 54 	 * @lends jQuery.operatingload
 55 	 */
 56 	{
 57 		/**
 58 		 * 初始化，拼装数据和模版加载到容器
 59 		 * 
 60 		 * @param {Object} option 参数对象
 61 		 */
 62 		init: function(options) {
 63 			var _ = this;
 64 			_.opts = $.extend({
 65 				tipText: '字测试测试文字测', //提示文本
 66 				URL: '', //操作地址
 67 				param: '', //向后台传的参数
 68 				time:5000,//时间,多长时间X号出现
 69 				callback: function(state, json) {} //回调函数
 70 			}, options);
 71 			var load = _.creatDiv();
 72 			load.hide();
 73 			var _wid = load.width() + 127;
 74 			var _hid = load.height();
 75 			//获取浏览器的宽度
 76 			var clientWidth = $(window).width();
 77 			var clientHeight = $(window).height();
 78 			//alert(clientWidth);
 79 			load.css("margin-left", -_wid / 2 + "px");
 80 			load.css("margin-top", -_hid / 2 + "px");
 81 			//给当前元素绑定事件
 82 			_.el.bind('click', function() {
 83 				_.getJson(_.opts.URL, _.opts.param, _.opts.callback);
 84 			});
 85 		},
 86 		/** 
 87 		 * 构造页面显示div
 88 		 */
 89 		creatDiv: function() {
 90 			var _load = $("<div>").addClass("ui-load-action").appendTo($("body"));
 91 			$("<img>").attr("width", 32).attr("height", 32).attr("src", "img/ui-loading.gif").addClass("ui-load-img").appendTo(_load);
 92 			var _txt = $("<p>").addClass("ui-load-text").appendTo(_load).html(this.opts.tipText);
 93 			return _load; //返回一个对象
 94 		},
 95 		/** 
 96 		 * 显示加载
 97 		 */
 98 		start: function() {
 99 			var load = $(".ui-load-action");
100 			if (load.length <= 0) {
101 				this.init();
102 			}
103 			load.show();
104 			setTimeout(function() {
105 				var _close = $("<img>").attr("width", 20).attr("height", 20).attr("src", "img/close.png").addClass("ui-load-close").appendTo(load);
106 				_close.bind("click", function() {
107 					load.fadeOut();
108 					$(this).remove();
109 				});
110 			}, this.opts.time);
111 		},
112 		/** 
113 		 * 关闭加载窗口
114 		 */
115 		done: function() {
116 			var load = $(".ui-load-action");
117 			load.fadeOut();
118 			$('.ui-load-close').remove();
119 		},
120 		/** 
121 		 * 从服务器获取数据，并提供给回调函数执行
122 		 * 
123 		 * @param {String} url - 操作访问的地址
124 		 * @param {String} param - 前台需要给后台的参数
125 		 * @param {Function} callback - 回调函数
126 		 * @return {Object} json对象
127 		 */
128 		getJson: function(url, param, callback) {
129 			var _ = this;
130 			_.start();
131 			$.ajax({
132 				url: url,
133 				type: "GET",
134 				data: "_=" + (new Date()).getTime() + (param == null || param == "" ? "" : ("&" + param)),
135 				cache: false,
136 				dataType: "json",
137 				beforeSend: function(xhr) {
138 					xhr.overrideMimeType("text/plain; charset=utf-8");
139 				},
140 				success: function(json) {
141 					_.done();
142 					callback('success', json);
143 				},
144 				error: function(e) {
145 					_.done();
146 					callback("error", {
147 						"flag": false,
148 						"errmsg": "调用服务失败！"
149 					});
150 				}
151 			});
152 		}
153 	}
154 })(jQuery)
