var aaa = $("#demo-radio").select({
	'type': 'radio',
	'name': 'radio',
	'content': [{
		"value": 'radio1',
		"text": "值1"
	},{
		"value": 'radio11',
		"text": "值11"
	},{
		"value": 'radio111',
		"text": "值111"
	},{
		"value": 'radio1111',
		"text": "值1111"
	},{
		"value": 'radio11111',
		"text": "值11111"
	}],
	'theme': 'ui-select-radio-blue-tradition',
	'flex':false
});



					
					

type ： 选择类型：单选或复选

theme ： 插件主题

content ： 选项的内容，包含每个选项的实际值value和展示值text

name ： 这组选择框的名称

flex ： 是否弹性布局，此项为false时：

flex ： 传统tradition和throughout的radio的宽度为100%；

flex ： 其他radio和checkbox的宽度为params.width；

width ： 选项的宽度，固定宽度的主题需要输入，默认200px

.select()-初始化方法

.getValue()——获取选择的实际值

.getText()——获取选择的显示值

.setValue()——设置某选项的实际值

.setText()——设置某选项的显示值

.setDisabled()——设置某选项为禁用

.setEnable()——设置某选项为可用
